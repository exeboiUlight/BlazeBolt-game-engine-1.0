🎯 Overview

BlazeBolt is a professional 2D game engine with Lua scripting support. All engine functions are accessible through the global BlazeBolt table.

💡 New in v2.0

  • Scene System - Load and switch between game scenes
  • End Callback - Cleanup when game ends
  • Multiple Windows - Create and manage additional windows
  • Object Deletion - Destroy entities at runtime
  • Texture Hot-Swapping - Change sprite textures dynamically
main.lua - Basic Game Structure 
-- Called once when game starts
function Start()
    BlazeBolt.Print("Game initialized!")
end

-- Called every frame (dt = delta time in seconds)
function Update(dt)
    if BlazeBolt.IsKeyPressed(Keys.ESCAPE) then
        BlazeBolt.Quit()
    end
end

-- Called after automatic drawing
function Draw()
    -- Custom drawing code here
end

-- Called when game ends (NEW!)
function End()
    BlazeBolt.Print("Game is shutting down...")
    -- Cleanup code here
end

📄 Scripts List Format (scripts.list)

Define your scripts and scenes in engine/scripts.list:

scripts.list 
# Regular scripts (loaded automatically)
main=scripts/main.lua
player=scripts/player.lua
ui=scripts/ui.lua

# Scenes (prefixed with @)
@menu=scenes/menu.lua
@game=scenes/game.lua
@settings=scenes/settings.lua

# Short form (name from filename)
scripts/enemies.lua

📌 Scene Script Structure

Each scene script should define:

  • On[SceneName]Load() - Called when scene loads
  • On[SceneName]Unload() - Called when scene unloads
  • Update(dt) - Scene update logic
  • Draw() - Scene drawing logic

🎬 Scene System Overview

Scenes allow you to organize your game into logical sections (menu, gameplay, settings, etc.)

scenes/menu.lua - Menu Scene Example 
local titleText, startButton

function OnMenuLoad()
    titleText = BlazeBolt.NewText("assets/font.ttf", "My Game", 400, 200, 64)
    startButton = BlazeBolt.NewSprite("assets/button.png", 400, 300)
    BlazeBolt.SpriteSetSize(startButton, 200, 50)
    BlazeBolt.Print("Menu loaded!")
end

function OnMenuUnload()
    BlazeBolt.Destroy(titleText)
    BlazeBolt.Destroy(startButton)
    BlazeBolt.Print("Menu unloaded!")
end

function Update(dt)
    local mx, my = BlazeBolt.GetMouseX(), BlazeBolt.GetMouseY()
    if BlazeBolt.IsMouseButtonJustPressed(MouseButtons.LEFT) then
        if mx > 400 and mx < 600 and my > 300 and my < 350 then
            BlazeBolt.LoadScene("game")
        end
    end
end
scenes/game.lua - Game Scene Example 
local player, score = 0, 0

function OnGameLoad()
    player = BlazeBolt.NewSprite("assets/player.png", 400, 300)
    BlazeBolt.SpriteSetSize(player, 64, 64)
    BlazeBolt.Print("Game started!")
end

function OnGameUnload()
    BlazeBolt.Destroy(player)
    BlazeBolt.Print("Game saved!")
end

function Update(dt)
    local x, y = BlazeBolt.SpriteGetPosition(player)
    local speed = 300 * dt
    
    if BlazeBolt.IsKeyPressed(Keys.A) then x = x - speed end
    if BlazeBolt.IsKeyPressed(Keys.D) then x = x + speed end
    if BlazeBolt.IsKeyPressed(Keys.W) then y = y - speed end
    if BlazeBolt.IsKeyPressed(Keys.S) then y = y + speed end
    
    BlazeBolt.SpriteSetPosition(player, x, y)
    
    if BlazeBolt.IsKeyJustPressed(Keys.ESCAPE) then
        BlazeBolt.LoadScene("menu")
    end
end

local player

    player = BlazeBolt.NewSprite("assets/idle.png", 400, 300)
    BlazeBolt.SpriteSetSize(player, 64, 64)
    BlazeBolt.SpriteSetOrigin(player, 0.5, 0.5)

player.lua - Player Controller Module 
local Player = {}

function Player.new(x, y)
    local self = {
        id = BlazeBolt.NewSprite("assets/player.png", x, y),
        speed = 300,
        hp = 100
    }
    BlazeBolt.SpriteSetSize(self.id, 64, 64)
    BlazeBolt.SpriteSetOrigin(self.id, 0.5, 0.5)
    return self
end

function Player:update(dt)
    local x, y = BlazeBolt.SpriteGetPosition(self.id)
    local move = self.speed * dt
    
    if BlazeBolt.IsKeyPressed(Keys.A) then x = x - move end
    if BlazeBolt.IsKeyPressed(Keys.D) then x = x + move end
    if BlazeBolt.IsKeyPressed(Keys.W) then y = y - move end
    if BlazeBolt.IsKeyPressed(Keys.S) then y = y + move end
    
    BlazeBolt.SpriteSetPosition(self.id, x, y)
end

return Player
game.lua - Game Scene 
local Player = require("scripts/player")
local player, score, scoreText

function OnGameLoad()
    player = Player.new(400, 300)
    score = 0
    scoreText = BlazeBolt.NewText("assets/font.ttf", "Score: 0", 10, 10, 32)
    BlazeBolt.Print("Game scene loaded!")
end

function OnGameUnload()
    BlazeBolt.Destroy(player.id)
    BlazeBolt.Destroy(scoreText)
    BlazeBolt.Print("Game scene unloaded!")
end

function Update(dt)
    player:update(dt)
    
    score = score + 1
    BlazeBolt.TextSetString(scoreText, "Score: " .. score)
    
    if BlazeBolt.IsKeyJustPressed(Keys.ESCAPE) then
        BlazeBolt.LoadScene("menu")
    end
end
