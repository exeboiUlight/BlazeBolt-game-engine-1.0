Система координат

OpenGL NDC: центр (0,0), правый верх (1,1), левый низ (-1,-1).

pixelsToNDC

function pixelsToNDC(px, py)
ПараметрТипОписание
pxnumberПиксель X
pynumberПиксель Y
Возвращает: ndcX, ndcY (number, number)
function pixelsToNDC(px, py)
    local ndcX = (px / GetScreenWidth()) * 2 - 1
    local ndcY = 1 - (py / GetScreenHeight()) * 2
    return ndcX, ndcY
end

ВНИМАНИЕ: World/Entity НЕ РЕАЛИЗОВАНЫ в Lua.

World/Entity

Пул объектов с авто-управлением памятью. Только C++.

Пример: мини-игра

local playerSprite, playerBody, score = 0, 0, 0
local scoreText, coins, coinBodies

function Start()
    playerSprite = BlazeBolt.CreateSprite("player.png", 0, 0)
    BlazeBolt.SpriteSetSize(playerSprite, 0.08, 0.08)
    playerBody = BlazeBolt.PhysicsCreateBody(1, 0, 0, 1, 0.2, 0.6)
    BlazeBolt.PhysicsAddRectangle(playerBody, 0.04, 0.04)

    local g = BlazeBolt.CreateSprite("ground.png", 0, -0.85)
    BlazeBolt.SpriteSetSize(g, 3, 0.05)
    BlazeBolt.PhysicsCreateBody(0, 0, -0.85, 0, 0.5, 0.3)

    scoreText = BlazeBolt.CreateText("fonts/arial.ttf", "Score: 0", -0.8, 0.85)
    BlazeBolt.TextSetColor(scoreText, 1, 1, 1, 1)

    BlazeBolt.LoadSound("music/bgm.mp3", "bgm", true)
    BlazeBolt.LoadSound("sfx/jump.wav", "jump", false)
    BlazeBolt.PlaySound("bgm")

    BlazeBolt.PhysicsInit(0, -9.8)
    spawnCoins()
end

function spawnCoins()
    coins = {}; coinBodies = {}
    for i = 1, 5 do
        local cx, cy = BlazeBolt.Random(-0.6, 0.6), BlazeBolt.Random(-0.5, 0.6)
        local c = BlazeBolt.CreateSprite("coin.png", cx, cy)
        BlazeBolt.SpriteSetSize(c, 0.06, 0.06)
        local cb = BlazeBolt.PhysicsCreateBody(0, cx, cy, 0, 0, 0)
        BlazeBolt.PhysicsAddCircle(cb, 0.03, 0, 0)
        table.insert(coins, c); table.insert(coinBodies, cb)
    end
end

function Update(dt)
    local vx, vy = BlazeBolt.PhysicsGetLinearVelocity(playerBody)
    if BlazeBolt.IsKeyPressed(Keys.A) then BlazeBolt.PhysicsApplyForce(playerBody, -15, 0) end
    if BlazeBolt.IsKeyPressed(Keys.D) then BlazeBolt.PhysicsApplyForce(playerBody, 15, 0) end
    if BlazeBolt.IsKeyJustPressed(Keys.SPACE) and vy < 0.1 then
        BlazeBolt.PhysicsApplyImpulse(playerBody, 0, 8)
        BlazeBolt.PlaySound("jump")
    end
    BlazeBolt.PhysicsStep()
    BlazeBolt.PhysicsSyncSprite(playerBody, playerSprite)
    local px, py = BlazeBolt.PhysicsGetPosition(playerBody)
    for i = #coins, 1, -1 do
        local cx, cy = BlazeBolt.PhysicsGetPosition(coinBodies[i])
        if math.sqrt((px-cx)^2 + (py-cy)^2) < 0.08 then
            BlazeBolt.Destroy(coins[i]); BlazeBolt.PhysicsDestroyBody(coinBodies[i])
            table.remove(coins, i); table.remove(coinBodies, i)
            score = score + 10
            BlazeBolt.TextSetString(scoreText, "Score: " .. score)
            BlazeBolt.PlaySound("collect")
        end
    end
    if #coins == 0 then spawnCoins() end
    if BlazeBolt.IsKeyJustPressed(Keys.ESCAPE) then BlazeBolt.Quit() end
end

Лучшие практики

  • Создавайте объекты в Start(), не в Update()
  • Destroy() удаляет ненужные объекты
  • Проектируйте в NDC — работает на любом разрешении
  • Для pixel-art используйте Nearest-фильтр (по умолчанию)
  • Перезагружайте скрипты через ReloadAllScripts без перезапуска
  • Проверяйте entity ~= 0 после создания (0 = ошибка)
  • Звуки: проверяйте soundId >= 0
